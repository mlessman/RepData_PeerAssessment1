data10 <- data2
data10$dayType <- "NA"
for(i in 1:nrow(data10)){
        if(data10$day[i]== "Saturday" | data10$day[i]== "Sunday"){
                data10$dayType[i] <- "Weekend"
        }else{
                data10$dayType[i] <- "Weekday"  
        }
}
